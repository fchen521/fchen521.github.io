##### Jquery 前端知识记录

##### Jquery动态创建表单
 
```ruby
<body>
  /* 添加表单按钮 */
  <button onclick="addHtml()">add</button>
  <div id="log">
  <h3>表单:</h3>
                /* 模拟数据回显存在的表单 */
		<div>
			<input/> <input/> <button class="btn row_data">remove</button>
	        </div>
		<div id='div'>
			<input/> <input/> <button class="btn row_data">remove</button>
		</div>
		<div>
			<input/><input/><button class="btn row_data">remove</button>
		</div>
	</div>
<script>
	var num = 1;
	function addHtml(){
		num++;
		var $log = $( "#log" );
                /* 要生成表单 */
		str = "<div id='div"+num+"'><input/> <input/> <button onclick='removeHtml("+num+")'>remove</button></div>";
		var html = $.parseHTML( str );
		$log.append( html );
	}
	function removeHtml(id){
		$("#div"+id).remove();
	}
	
        /* 通过按钮的class属性，定位选择要删除的表单 */
	$(".row_data").click(function(e){
		var i = $(".row_data").index($(this));
		$(".row_data").eq(i).parent().remove();
	})
</script>
</body>
```
